    // src/features/calendar/selectEventsForCalendar.ts
    // Pure adapter: ScheduleEvent[] + Task[] → FullCalendar EventInput[].
    // Single point where domain shapes meet FullCalendar — keeps FullCalendar concerns out of domain.
    //
    // Source: [CITED: 03-CONTEXT.md D-24]
    //         [CITED: 03-RESEARCH.md §Pattern 6 + §Pitfall 5 (FullCalendar end is EXCLUSIVE)]
    //         [CITED: 03-PATTERNS.md §selectEventsForCalendar.ts]
    //
    // Purity: zero React/Zustand/I/O; date math via dateWrappers only.

    import type { ScheduleEvent, Task, EventType } from '../../domain/types';
    import { lifecyclePalette } from '../gantt/lifecyclePalette';
    import { parseDate, addDays, toISODate } from '../../domain/dateWrappers';

    export interface CalendarEventInput {
      id: string;
      title: string;
      start: string;          // 'YYYY-MM-DD'
      end?: string;           // 'YYYY-MM-DD', EXCLUSIVE per FullCalendar contract
      backgroundColor?: string;
      borderColor?: string;
      extendedProps: {
        kind: 'lifecycle' | 'task';
        eventType: string;
        plantingId?: string;
        taskId?: string;
      };
    }

    // Auto-task event types — these render via the Task[] parameter, not as lifecycle bars.
    const AUTO_TASK_TYPES: ReadonlySet<EventType> = new Set([
      'water-seedlings',
      'harden-off-day',
      'fertilize-at-flowering',
    ]);

    function humanLabel(type: EventType): string {
      // Convert kebab-case event types to "Title Case" for display.
      return type.split('-').map((w) => w[0].toUpperCase() + w.slice(1)).join(' ');
    }

    function exclusiveEnd(endISO: string): string {
      // FullCalendar end is the date AFTER the last day. Add one day per Pitfall 5.
      return toISODate(addDays(parseDate(endISO), 1)).slice(0, 10);
    }

    export function selectEventsForCalendar(
      events: ScheduleEvent[],
      tasks: Task[],
    ): CalendarEventInput[] {
      const out: CalendarEventInput[] = [];

      // Lifecycle anchors + ranges
      for (const e of events) {
        if (AUTO_TASK_TYPES.has(e.type)) continue;
        const fill = lifecyclePalette[e.type];
        if (!fill) continue; // defensive — palette omission means "do not render"

        const start = e.start.slice(0, 10);
        const end = e.end !== e.start ? exclusiveEnd(e.end) : undefined;

        out.push({
          id: e.id,
          title: humanLabel(e.type),
          start,
          end,
          backgroundColor: fill,
          borderColor: fill,
          extendedProps: { kind: 'lifecycle', eventType: e.type, plantingId: e.plantingId },
        });
      }

      // Tasks (auto + custom + recurring expanded — supplied by caller from Plan 03-05's deriveTasks)
      for (const t of tasks) {
        out.push({
          id: `task:${t.id}`,
          title: t.title,
          start: t.dueDate.slice(0, 10),
          backgroundColor: '#f5f5f4', // stone-100 — neutral; differentiates from lifecycle
          borderColor: '#a8a29e',     // stone-400
          extendedProps: { kind: 'task', eventType: 'task', taskId: t.id, plantingId: t.plantingId },
        });
      }

      return out;
    }
    ```

    // src/features/calendar/useDayDetailUrl.ts
    // useSearchParams wrapper for ?date= state. Drawer open/closed derived from URL presence.
    // Per CONTEXT D-29 + RESEARCH.md §Pattern 8.

    import { useSearchParams } from 'react-router';

    export interface DayDetailUrlState {
      selectedDate: string | null; // 'YYYY-MM-DD' or null
      isOpen: boolean;
      open: (dateStr: string) => void;
      close: () => void;
    }

    export function useDayDetailUrl(): DayDetailUrlState {
      const [params, setParams] = useSearchParams();
      const selectedDate = params.get('date');
      const isOpen = selectedDate !== null;

      function open(dateStr: string) {
        const next = new URLSearchParams(params);
        next.set('date', dateStr);
        setParams(next, { replace: false }); // pushState — back-button closes
      }

      function close() {
        const next = new URLSearchParams(params);
        next.delete('date');
        setParams(next, { replace: false });
      }

      return { selectedDate, isOpen, open, close };
    }
    ```

    // src/features/calendar/DayDetailDrawer.tsx
    // Radix Dialog right-side sheet — events + tasks for the URL-selected day.
    // Per CONTEXT D-29..D-31 + UI-SPEC §7.
    // Source: [CITED: 03-PATTERNS.md §DayDetailDrawer.tsx]
    //         [CITED: src/features/catalog/MyPlanPanel.tsx lines 78-103 (exact pattern)]

    import * as DialogPrimitive from '@radix-ui/react-dialog';
    import { X } from 'lucide-react';
    import { format } from 'date-fns';
    import { useDayDetailUrl } from './useDayDetailUrl';
    import { useDerivedSchedule } from '../gantt/useDerivedSchedule';
    import { useCatalogStore, selectMerged } from '../../stores/catalogStore';
    import { usePlanStore } from '../../stores/planStore';
    import { Dialog, DialogPortal, DialogOverlay, DialogTitle } from '../../ui/Dialog';
    import { lifecyclePalette } from '../gantt/lifecyclePalette';
    import { cn } from '../../ui/cn';
    import { parseDate } from '../../domain/dateWrappers';

    export function DayDetailDrawer() {
      const { selectedDate, isOpen, close } = useDayDetailUrl();
      const events = useDerivedSchedule();
      const catalog = useCatalogStore(selectMerged);
      const plan = usePlanStore((s) => s.plan);

      if (!selectedDate) return null;

      // Filter events to the selected day (UTC date comparison via slice).
      const dayEvents = events.filter((e) => {
        const eStart = e.start.slice(0, 10);
        const eEnd = e.end.slice(0, 10);
        return selectedDate >= eStart && selectedDate <= eEnd;
      });

      // Group by plantingId
      const groups = new Map<string, typeof dayEvents>();
      for (const e of dayEvents) {
        const arr = groups.get(e.plantingId) ?? [];
        arr.push(e);
        groups.set(e.plantingId, arr);
      }

      // Tasks for the day come from Plan 03-05's deriveTasks; for THIS plan, tasks are not yet wired.
      // Pass empty array placeholder; Plan 03-07 wires the real Task[] consumer.
      const dayTasks: Array<{ id: string; title: string; plantingId?: string }> = [];

      const heading = (() => {
        try {
          return format(parseDate(`${selectedDate}T12:00:00.000Z`), 'EEEE, LLLL d, yyyy');
        } catch {
          return selectedDate;
        }
      })();

      const eventCount = dayEvents.length;
      const taskCount = dayTasks.length;
      const subhead = [
        eventCount > 0 ? `${eventCount} event${eventCount === 1 ? '' : 's'}` : null,
        taskCount > 0 ? `${taskCount} task${taskCount === 1 ? '' : 's'}` : null,
      ].filter(Boolean).join(' · ');

      const isEmpty = eventCount === 0 && taskCount === 0;

      return (
        <Dialog open={isOpen} onOpenChange={(o) => { if (!o) close(); }}>
          <DialogPortal>
            <DialogOverlay className="fixed inset-0 z-40 bg-stone-900/40 backdrop-blur-sm" />
            <DialogPrimitive.Content
              aria-labelledby="day-detail-heading"
              className={cn(
                'fixed right-0 top-0 z-50 h-full w-[var(--spacing-drawer-w)] max-w-[calc(100vw-32px)]',
                'bg-white shadow-[-8px_0_24px_rgb(0_0_0_/_0.08)] border-l border-stone-200 overflow-y-auto',
                'transition-transform duration-200 ease-out',
                'data-[state=open]:translate-x-0 data-[state=closed]:translate-x-full',
                'focus:outline-none',
              )}
            >
              <div className="flex items-center justify-between sticky top-0 bg-white px-6 pt-6 pb-3 border-b border-stone-100">
                <div>
                  <DialogTitle id="day-detail-heading" className="text-xl font-semibold text-stone-900">
                    {heading}
                  </DialogTitle>
                  {subhead && <p className="text-sm font-normal text-stone-600 mt-1">{subhead}</p>}
                </div>
                <DialogPrimitive.Close asChild>
                  <button
                    type="button"
                    aria-label="Close"
                    className="text-stone-500 hover:text-stone-900 focus-visible:outline-2 focus-visible:outline-offset-2 focus-visible:outline-green-700"
                  >
                    <X className="h-5 w-5" />
                  </button>
                </DialogPrimitive.Close>
              </div>

              <div className="px-6 py-4">
                {isEmpty ? (
                  <div className="py-12 text-center">
                    <p className="text-xl font-semibold text-stone-900 mb-2">Nothing scheduled</p>
                    <p className="text-base text-stone-600">
                      No lifecycle events or tasks for {format(parseDate(`${selectedDate}T12:00:00.000Z`), 'EEE, LLL d')}.
                      Click another day to inspect, or close this panel.
                    </p>
                  </div>
                ) : (
                  Array.from(groups.entries()).map(([plantingId, items]) => {
                    const planting = plan?.plantings.find((p) => p.id === plantingId);
                    const plant = planting ? catalog.get(planting.plantId) : undefined;
                    const groupName = plant?.name ?? planting?.label ?? plantingId;
                    return (
                      <div key={plantingId} className="mt-4 mb-2">
                        <h3 className="text-sm font-semibold uppercase tracking-wider text-stone-900 mb-2">
                          {groupName}
                        </h3>

    export function PlanViewTabs() {
      const [searchParams, setSearchParams] = useSearchParams();
      const view = searchParams.get('view') === 'calendar' ? 'calendar' : 'gantt';

      function setView(next: 'gantt' | 'calendar') {

        if (next === 'gantt') sp.delete('view');

          {TABS.map(({ id, label, Icon }) => {
            const active = view === id;
            return (
              <button
                key={id}
                type="button"
                role="tab"
                aria-selected={active}
                onClick={() => setView(id)}
                className={cn(

          })}

## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| URL `?date=YYYY-MM-DD` → DayDetailDrawer | Untrusted URL value flows into `parseDate(...)` for the heading format and into `selectedDate` comparison. Malformed values get rendered as the raw string in heading + "no events for this day" empty state. |
| URL `?view=` → PlanViewTabs | Two-value enum (`gantt` | `calendar`); any other value defaults to `gantt` per the ternary `=== 'calendar' ? 'calendar' : 'gantt'`. |
| FullCalendar event objects → DOM render | FullCalendar handles its own escaping; titles flow through `selectEventsForCalendar.humanLabel` which is built from EventType union members (no user input). Tasks pass `t.title` through — Plan 03-05's CustomTaskModal is responsible for input validation. |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-03-04-01 | Tampering | Crafted `?date=2026-99-99` URL crashes the drawer's `parseDate` | mitigate | DayDetailDrawer wraps `format(parseDate(...))` in a try/catch; on parse error renders the raw `selectedDate` string in heading. The events-filter comparison still works because it's purely string-prefix matching on `selectedDate`. |
| T-03-04-02 | Tampering | Crafted `?view=` value (e.g. `?view=admin`) | accept | Default-to-gantt ternary covers; no admin/privilege concept exists. |
| T-03-04-03 | Information Disclosure | Drawer reveals task data (e.g. completion notes) | accept | Single-user local app; no PII boundary. |
| T-03-04-04 | Denial of Service | FullCalendar fails to lazy-load (network error after deploy) | mitigate | Plan 03-07 wires `<Suspense>` + ErrorBoundary fallback per Pitfall 6; this plan ships CalendarView as default export so the lazy boundary is well-defined. The error UI ("Calendar didn't load — switch back to Gantt") is wired in Plan 03-07. |
| T-03-04-05 | Spoofing | A malicious link `https://app/?view=calendar&date=...` lands the user in an unintended state | accept | URL-as-state is a feature (deep-linking); the worst outcome is the drawer opens to a strange day. Single-user local app — no XSRF surface. |
