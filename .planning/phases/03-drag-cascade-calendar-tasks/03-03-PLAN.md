<objective>
Make the gantt interactive. Wraps the existing read-only `<GanttView>` in a
`<DndContext>` provider, adds per-bar `useDraggable` via `useDragBar`, runs the
constraint registry inside a custom `@dnd-kit` modifier so clamp + reasons are
computed every tick (no second round-trip), renders a live ghost cascade overlay
via `<DragOverlay>` + `useTransientSchedule`, and surfaces constraint violations
as a portaled `ConstraintTooltip` (floating-with-cursor → sticky pill). Commits
the transient edit to `planStore.commitEdit` on pointerup, which zundo records as
ONE history entry.

Purpose: Closes GANTT-04 (drag), GANTT-05 (constraint enforcement), GANTT-06
(cascade preview), GANTT-07 (commit), and GANTT-09 (snap-back tooltip). Lock UI
+ undo/redo header buttons + AppShell wiring are deferred to Plan 03-06 (Wave 3)
because they touch GanttView.tsx + AppShell.tsx, sequencing after this plan.

Output:
- Drag mechanics for indoor-start / transplant / direct-sow whole-bar moves
- Right-edge resize for harvest-window
- Cascade ghost preview that visually previews exactly what commit will produce
- Constraint clamp + portaled tooltip explaining "why" on snap-back
- Sparse `ScheduleEdit` commit pipeline through zundo for undo/redo

    // rAF-throttled selector that runs the live cascade preview during drag.
    // Source: [CITED: 03-CONTEXT.md D-19, D-20]
    //         [CITED: 03-RESEARCH.md §Pattern 3]
    //         [CITED: 03-PATTERNS.md §useTransientSchedule.ts]

    import { useMemo } from 'react';
    import { usePlanStore } from '../../../stores/planStore';
    import { useCatalogStore, selectMerged } from '../../../stores/catalogStore';
    import { useDragStore } from '../../../stores/dragStore';
    import { generateScheduleWithLocks } from '../../../domain/schedulerWithLocks';
    import { expandSuccessions } from '../../../domain/succession';
    import type { ScheduleEvent } from '../../../domain/types';

    export function useTransientSchedule(): ScheduleEvent[] {
      const plan = usePlanStore((s) => s.plan);
      const catalog = useCatalogStore(selectMerged);
      const transientEdit = useDragStore((s) => s.transientEdit);
      return useMemo<ScheduleEvent[]>(() => {
        if (!plan) return [];
        // Apply transient edit IN PLACE on the edits array; engine consumes it.
        // Dedupe by (plantingId, eventType) — last write wins (mirrors planStore.commitEdit dedupe).
        const planWithTransient = transientEdit
          ? {
              ...plan,
              edits: [
                ...plan.edits.filter(
                  (e) => !(e.plantingId === transientEdit.plantingId && e.eventType === transientEdit.eventType),
                ),
                transientEdit,
              ],
            }
          : plan;
        const expanded = expandSuccessions(planWithTransient, catalog);
        return generateScheduleWithLocks(expanded, catalog);
      }, [plan, catalog, transientEdit]);
    }

    // src/features/gantt/drag/clampModifier.ts
    // @dnd-kit modifier — runs every drag tick BEFORE the bar visually moves.
    // Calls canMove(); if clamped, returns the clamped transform AND writes reasons[] to dragStore.
    // Per CONTEXT D-08.
    //
    // Source: [CITED: 03-CONTEXT.md D-07, D-08]
    //         [CITED: 03-RESEARCH.md §Pattern 1]
    //         [CITED: dndkit.com/api-documentation/modifiers]
    //
    // Purity: pure factory; the returned modifier closes over plan/plant/scale references.

    import type { Modifier } from '@dnd-kit/core';
    import type { TimeScale } from '../timeScale';
    import type { GardenPlan, Plant, ScheduleEvent, EventType } from '../../../domain/types';
    import { canMove } from '../../../domain/constraints';
    import { parseDate, addDays, toISODate } from '../../../domain/dateWrappers';

    interface ClampDeps {
      scale: TimeScale;
      event: ScheduleEvent;
      plan: GardenPlan;
      plant: Plant;
      setViolation: (v: { eventId: string; eventType: EventType; reasons: string[] } | null) => void;
    }

    export function makeClampModifier(deps: ClampDeps): Modifier {
      return ({ transform, activeNodeRect }) => {
        if (!activeNodeRect) return transform;

        // Whole-day snap: convert pixel delta → day delta → date → back to px.
        const startX = deps.scale.dateToX(deps.event.start);
        const candidateX = startX + transform.x;
        const candidateDate = deps.scale.xToDate(candidateX); // returns ISO; xToDate already day-snaps per timeScale contract

        const result = canMove(deps.event, candidateDate, deps.plan, deps.plant);

        if ('clamped' in result && result.clamped) {
          deps.setViolation({
            eventId: deps.event.id,
            eventType: deps.event.type,
            reasons: result.reasons,
          });
          const clampedX = deps.scale.dateToX(result.finalDate);
          return { ...transform, x: clampedX - startX };
        }

        // Pass-through: clear violation if previously set; snap to whole-day.
        deps.setViolation(null);
        const snappedX = deps.scale.dateToX(candidateDate);
        return { ...transform, x: snappedX - startX };
      };
    }
    ```

    // src/features/gantt/drag/useDragBar.ts
    // Per-bar useDraggable wrapper. Memoizes the modifier so React only re-renders bars whose
    // dates actually change during drag (per CONTEXT D-21 + RESEARCH §Pitfall 1).
    //
    // For non-draggable event types (D-06), returns no-op listeners so the bar is still clickable
    // for select/inspect + lock-toggle hover.

    import { useDraggable } from '@dnd-kit/core';
    import { useMemo } from 'react';
    import type { GardenPlan, Plant, ScheduleEvent } from '../../../domain/types';
    import type { TimeScale } from '../timeScale';
    import { makeClampModifier } from './clampModifier';
    import { useDragStore } from '../../../stores/dragStore';

    const DRAGGABLE_TYPES = new Set(['indoor-start', 'transplant', 'direct-sow', 'harvest-window']);

    export function useDragBar(args: {
      event: ScheduleEvent;
      plan: GardenPlan;
      plant: Plant;
      scale: TimeScale;
    }) {
      const setViolation = useDragStore((s) => s.setLastConstraintViolation);
      const isDraggableType = DRAGGABLE_TYPES.has(args.event.type);

      const modifiers = useMemo(
        () =>
          isDraggableType
            ? [makeClampModifier({
                scale: args.scale,
                event: args.event,
                plan: args.plan,
                plant: args.plant,
                setViolation,
              })]
            : [],
        // Memoize on the inputs that change drag math; per Pitfall 1 + D-21
        [
          isDraggableType,
          args.scale,
          args.event.id,
          args.event.start,
          args.event.end,
          args.event.edited,
          args.plan.location.lastFrostDate,
          args.plan.location.firstFrostDate,
          args.plant.id,
          setViolation,
        ],
      );

      const { attributes, listeners, setNodeRef, transform, isDragging } = useDraggable({
        id: args.event.id,
        disabled: !isDraggableType,
        data: { event: args.event, plant: args.plant }, // available on active.data.current in DragLayer
      });

      return {
        attributes,
        listeners: isDraggableType ? listeners : undefined,
        setNodeRef,
        transform,
        isDragging,
        modifiers,
      };
    }
    ```

    import { makeClampModifier } from './clampModifier';
    import { useTransientSchedule } from './useTransientSchedule';
    import type { GardenPlan, Plant, ScheduleEvent, EventType, ScheduleEdit } from '../../../domain/types';
    import { useTimeScale } from '../useTimeScale'; // verify this export exists; if not, derive scale here from useDerivedSchedule
    import { nowISOString } from '../../../domain/dateWrappers';

    export function DragLayer() {
      const sensors = useSensors(useSensor(PointerSensor, { activationConstraint: { distance: 4 } }));
      const plan = usePlanStore((s) => s.plan);
      const catalog = useCatalogStore(selectMerged);
      const setViolation = useDragStore((s) => s.setLastConstraintViolation);
      const setUiViolation = useUIStore((s) => s.setLastConstraintViolation);
      const beginDrag = useDragStore((s) => s.beginDrag);
      const setTransientEdit = useDragStore((s) => s.setTransientEdit);
      const endDrag = useDragStore((s) => s.endDrag);
      const commitEdit = usePlanStore((s) => s.commitEdit);
      const transientSchedule = useTransientSchedule();
      const activeIdRef = useRef<string | null>(null);

      // Single dispatcher modifier: reads active.data.current to find the event/plant for the active drag.
      // Memoized on (plan, catalog) so React identity is stable across drag ticks.
      const dispatcherModifier: Modifier = useMemo(() => {
        return (args) => {
          if (!plan) return args.transform;
          const active = args.active;
          if (!active) return args.transform;
          const data = active.data.current as { event?: ScheduleEvent; plant?: Plant } | undefined;
          if (!data?.event || !data.plant) return args.transform;
          // Lazily build a per-event clamp modifier each tick (cheap; closes over fresh plan/scale).
          // The TimeScale must be retrievable here; we read it from a single shared module-level scale
          // (see GanttView extraction note in step C below).
          const scale = (window as any).__gardenGanttTimeScale;
          if (!scale) return args.transform;
          const inner = makeClampModifier({ scale, event: data.event, plan, plant: data.plant, setViolation });
          return inner(args);
        };
      }, [plan, setViolation]);

      const handleDragStart = useCallback((e: DragStartEvent) => {
        activeIdRef.current = String(e.active.id);
        beginDrag(String(e.active.id));
        // Clear any prior sticky-pill from previous drag.
        setUiViolation(null);
      }, [beginDrag, setUiViolation]);

      const handleDragMove = useCallback((e: { active: { id: string | number }; delta: { x: number; y: number } }) => {
        // Build the transient edit from the active event + delta, then write to dragStore.
        // The clampModifier already wrote violation reasons; here we just build the edit shape.
        if (!plan) return;
        const data = (e as any).active?.data?.current as { event?: ScheduleEvent; plant?: Plant } | undefined;
        if (!data?.event) return;
        const scale = (window as any).__gardenGanttTimeScale;
        if (!scale) return;
        const startX = scale.dateToX(data.event.start);
        const newDate = scale.xToDate(startX + e.delta.x);
        const edit: ScheduleEdit = {
          plantingId: data.event.plantingId,
          eventType: data.event.type,
          startOverride: newDate,
          endOverride: data.event.end !== data.event.start
            ? scale.xToDate(scale.dateToX(data.event.end) + e.delta.x)
            : undefined,
          reason: 'user-drag',
          editedAt: nowISOString(),
        };
        // rAF coalesce: schedule the setTransientEdit on the next frame; planStore handleSet will dedupe.
        scheduleRafSet(setTransientEdit, edit);
      }, [plan, setTransientEdit]);

      const handleDragEnd = useCallback((e: DragEndEvent) => {
        const transientEdit = useDragStore.getState().transientEdit;
        const violation = useDragStore.getState().lastConstraintViolation;
        if (transientEdit) {
          commitEdit(transientEdit); // ONE history entry via zundo handleSet rAF debounce
        }
        // If last frame was clamped, promote the dragStore violation to uiStore (sticky-pill phase).
        if (violation) {
          useUIStore.getState().setLastConstraintViolation(violation);
        }
        endDrag();
        activeIdRef.current = null;
      }, [commitEdit, endDrag]);

      const handleDragCancel = useCallback(() => {
        endDrag();
        activeIdRef.current = null;
      }, [endDrag]);

      const activeId = useDragStore((s) => s.activeEventId);

      return (
        <DndContext
          sensors={sensors}
          modifiers={[dispatcherModifier]}
          onDragStart={handleDragStart}
          onDragMove={handleDragMove}
          onDragEnd={handleDragEnd}
          onDragCancel={handleDragCancel}
        >
          <GanttView />
          {/* Per dnd-kit docs: remain mounted at all times. */}
          <DragOverlay dropAnimation={null}>
            {/* The actively-dragged bar visual content lives here at full opacity, follows cursor. */}
            {/* Phase 3 ships a minimal placeholder; visual fidelity polish in Phase 4 if needed. */}
            {activeId ? <ActiveBarOverlay activeId={activeId} events={transientSchedule} /> : null}
          </DragOverlay>
        </DndContext>
      );
    }

    // rAF coalesce helper — collapse 60Hz pointermove into one update per frame.
    let rafId: number | null = null;
    let pendingEdit: ScheduleEdit | null = null;
    function scheduleRafSet(setter: (e: ScheduleEdit | null) => void, edit: ScheduleEdit) {
      pendingEdit = edit;
      if (rafId !== null) return;
      rafId = requestAnimationFrame(() => {
        rafId = null;
        if (pendingEdit) setter(pendingEdit);
      });
    }

    // Minimal placeholder for the actively-dragged bar in the DragOverlay portal.
    // Renders a small label following the cursor; Phase 4 may substitute a faithful bar render.
    function ActiveBarOverlay({ activeId, events }: { activeId: string; events: ScheduleEvent[] }) {
      const e = events.find((x) => x.id === activeId);
      if (!e) return null;
      return (
        <div className="px-2 py-1 rounded-md bg-stone-900 text-white text-xs shadow-lg pointer-events-none">
          {e.type} → {e.start.slice(0, 10)}
        </div>
      );
    }
