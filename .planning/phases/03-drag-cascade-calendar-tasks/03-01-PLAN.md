       ```typescript
       {
         name: 'hardenOffMustPrecedeTransplant',
         appliesTo: (e) => e.type === 'transplant',
         check: (event, candidate, plan, plant) => {
           if (!plant.timing.requiresHardening) return { ok: true, finalDate: candidate, reasons: [] };
           const hardenDays = plant.timing.daysToHardenOff ?? 7;

           // GANTT-05 invariant: harden-off must precede transplant.
           // Implied schedule when transplant lands on `candidate`:
           //   hardenOffEnd   = candidate - 1 day
           //   hardenOffStart = hardenOffEnd - hardenDays
           // The earliest legitimate hardenOffStart is the indoor anchor (when seedlings exist):
           //   indoorAnchor = max(today, planting.indoorStartDate ?? today)
           // (`today` is approximated via plan.location.lastFrostDate's year + nowISOString — the domain
           // is pure, so we cannot truly read wall-clock; instead compare against the planting's
           // indoor-start anchor from plan.edits[] if present, else the computed indoor-start from
           // plant.timing.indoorStartOffsetDaysFromLastFrost.)
           const planting = plan.plantings.find((p) => p.id === event.plantingId);
           if (!planting) return { ok: true, finalDate: candidate, reasons: [] };

           // Find the indoor-start anchor (post-edit if present)
           const indoorEdit = plan.edits.slice().reverse().find(
             (ed) => ed.plantingId === event.plantingId && ed.eventType === 'indoor-start'
           );
           const baseLastFrost = parseDate(plan.location.lastFrostDate);
           const computedIndoorStart = toISODate(addDays(
             baseLastFrost,
             plant.timing.indoorStartOffsetDaysFromLastFrost ?? -49
           ));
           const indoorAnchor = indoorEdit ? indoorEdit.startOverride : computedIndoorStart;

           const candidateDate = parseDate(candidate);
           const hardenOffEnd = addDays(candidateDate, -1);
           const hardenOffStart = addDays(hardenOffEnd, -hardenDays);
           const indoorAnchorDate = parseDate(indoorAnchor);

           if (hardenOffStart.getTime() >= indoorAnchorDate.getTime()) {
             return { ok: true, finalDate: candidate, reasons: [] };
           }

           // Clamp candidate forward so harden-off can begin at indoorAnchor:
           //   minTransplant = indoorAnchor + hardenDays + 1 day
           const minTransplant = toISODate(addDays(indoorAnchorDate, hardenDays + 1));
           return {
             ok: true,
             clamped: true,
             finalDate: minTransplant,
             reasons: [`Harden-off must precede transplant by ${hardenDays} days (earliest transplant: ${minTransplant.slice(0, 10)}).`],
           };
         },
       },
       ```
