    ```typescript
    /**
     * @vitest-environment happy-dom
     * Phase 2 Flow A integration test — the canonical happy path.
     */
    import { describe, it, expect, beforeEach, vi } from 'vitest';
    import { render, screen, waitFor } from '@testing-library/react';
    import userEvent from '@testing-library/user-event';
    import { MemoryRouter } from 'react-router';
    import App from '../../src/app/App';  // verify export shape — App may be default-exported as <App/> or named

    describe('Phase 2 Flow A — happy path', () => {
      beforeEach(() => {
        window.localStorage.clear();
        vi.resetModules();
        vi.spyOn(globalThis, 'fetch').mockImplementation(async (url) => {
          if (typeof url === 'string' && url.includes('/data/zones.2.json')) {
            return new Response(JSON.stringify({
              version: 1, generatedAt: 'x',
              zips: { '20001': { zone: '7a', lat: 38.9, lon: -77.0, lastSpringFrost50: '04-15', firstFallFrost50: '10-20' } },
            }));
          }
          return new Response('Not found', { status: 404 });
        });
      });

      it('full happy path: ZIP → plants → review → finish → reload preserves state', async () => {
        const user = userEvent.setup();

        // 1. Render app at /setup (initial state — plan === null)
        const { unmount } = render(
          <MemoryRouter initialEntries={['/setup']}><App /></MemoryRouter>
        );

        // 2. Step 1: enter ZIP, wait for derived block
        const zipInput = await screen.findByLabelText(/your zip code/i);
        await user.type(zipInput, '20001');
        await waitFor(() => expect(screen.getByText(/7a/i)).toBeInTheDocument(), { timeout: 2000 });

        // 3. Click Next → Step 2 (catalog browser)
        const next = screen.getByRole('button', { name: /next/i });
        await user.click(next);
        await waitFor(() => expect(screen.getByPlaceholderText(/search plants by name/i)).toBeInTheDocument());

        // 4. Add 5 plants
        const { usePlanStore } = await import('../../src/stores/planStore');
        const addButtons = await screen.findAllByRole('button', { name: /add .* to plan/i });
        for (let i = 0; i < 5 && i < addButtons.length; i++) {
          await user.click(addButtons[i]!);
        }
        await waitFor(() => expect(usePlanStore.getState().plan?.plantings.length).toBe(5));

        // 5. Click Next → Step 3 (review)
        // 6. Click Finish — go to my plan → /plan view
        // 7. Unmount, remount → state should persist via localStorage rehydrate
        unmount();
        render(<MemoryRouter initialEntries={['/plan']}><App /></MemoryRouter>);
        await waitFor(() => {
          const restored = usePlanStore.getState().plan;
          expect(restored?.plantings.length).toBe(5);
        });
      });
    });
    ```
