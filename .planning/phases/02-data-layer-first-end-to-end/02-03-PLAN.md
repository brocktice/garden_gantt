    ```typescript
    // cors-proxy/src/index.ts — Cloudflare Worker proxy for Permapeople API.
    // Stateless. No user data crosses this Worker — only generic plant lookups (D-17).
    // Source: [CITED: 02-RESEARCH.md §Pattern 5]

    interface Env {
      PERMAPEOPLE_KEY_ID: string;
      PERMAPEOPLE_KEY_SECRET: string;
      ALLOWED_ORIGIN: string;
    }

    export default {
      async fetch(req: Request, env: Env): Promise<Response> {
        const url = new URL(req.url);
        const origin = req.headers.get('Origin') ?? '';
        const allow = origin === env.ALLOWED_ORIGIN ? origin : env.ALLOWED_ORIGIN;
        const corsHeaders: Record<string, string> = {
          'Access-Control-Allow-Origin': allow,
          'Access-Control-Allow-Methods': 'GET, POST, OPTIONS',
          'Access-Control-Allow-Headers': 'Content-Type',
          'Vary': 'Origin',
        };
        if (req.method === 'OPTIONS') {
          return new Response(null, { status: 204, headers: corsHeaders });
        }
        const target =
          url.pathname === '/search'
            ? 'https://permapeople.org/api/search'
            : url.pathname.startsWith('/plants/')
              ? `https://permapeople.org/api${url.pathname}`
              : null;
        if (!target) return new Response('Not Found', { status: 404, headers: corsHeaders });

        const upstreamHeaders = new Headers(req.headers);
        upstreamHeaders.set('x-permapeople-key-id', env.PERMAPEOPLE_KEY_ID);
        upstreamHeaders.set('x-permapeople-key-secret', env.PERMAPEOPLE_KEY_SECRET);
        upstreamHeaders.delete('Origin');

        const upstream = await fetch(target, {
          method: req.method,
          headers: upstreamHeaders,
          body: req.method === 'POST' ? await req.text() : undefined,
        });
        const body = await upstream.text();
        return new Response(body, {
          status: upstream.status,
          headers: {
            ...corsHeaders,
            'Content-Type': upstream.headers.get('Content-Type') ?? 'application/json',
          },
        });
      },
    };
    ```

<threat_model>
## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| Browser → Cloudflare Worker | Browser may send any payload; Worker enforces ALLOWED_ORIGIN check + only allows /search and /plants/{id} paths. |
| Cloudflare Worker → Permapeople API | Worker injects API key from env (never browser-side). User data does NOT cross — only generic plant query strings. |
| Permapeople response → Worker → browser | Untrusted upstream content forwarded to browser. Browser-side searchPlant maps fields defensively (typeof checks). |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-02-09 | Spoofing | Browser → Worker | mitigate | ALLOWED_ORIGIN check in Worker rejects unknown origins. Worker is read-only (no auth, no user state to compromise). |
| T-02-10 | Tampering | Permapeople response → app state | mitigate | searchPlant's mapPermapeopleToEnrichment uses `typeof === 'string'` checks per field; only EnrichmentFields shape leaves the module. Consumer (CustomPlantModal in Plan 02-09) MUST treat description as text-only — display via React's default text-content escaping (no dangerouslySetInnerHTML). |
| T-02-11 | Repudiation | API key leak | mitigate | API key lives in `wrangler secret`; never in source code, .env.local, or browser bundle. .env.local.example shows VITE_PERMAPEOPLE_KEY_* commented out as DEV-only direct-mode fallback. |
| T-02-12 | Information disclosure | API key in browser bundle | mitigate | VITE_PERMAPEOPLE_KEY_ID/SECRET would only be present if dev set them; production deployment uses proxy path with no key in client. Verify via `grep -c VITE_PERMAPEOPLE_KEY` in built dist (Plan 02-12 verification). |
| T-02-13 | DoS | Worker quota exhaustion | accept | Cloudflare Workers free tier (100K req/day) covers single-user app comfortably (RESEARCH Assumption A5). Pay-as-you-go fallback is $0.50/M requests. No rate-limiting in Worker; deferred to follow-up if quota pressure surfaces. |
| T-02-14 | DoS | Hung requests blocking UI | mitigate | AbortController + 8s timeout in searchPlant. Tested. Modal close calls abort signal (wired in Plan 02-09). |
</threat_model>
