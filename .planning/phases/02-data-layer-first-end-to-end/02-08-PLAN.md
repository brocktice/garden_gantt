    ```typescript
    // src/features/setup/lookupLocation.ts
    // Hook wrapper around data/zones.ts lookupLocation. Memoizes per zip+year, cancels stale results.

    import { useEffect, useState } from 'react';
    import { lookupLocation, type LookupResult } from '../../data/zones';

    export type LookupHookState =
      | { status: 'idle' }
      | { status: 'loading' }
      | LookupResult;

    export function useLookupLocation(zip: string, year: number): LookupHookState {
      const [result, setResult] = useState<LookupHookState>({ status: 'idle' });

      useEffect(() => {
        if (!/^\d{5}$/.test(zip)) {
          setResult({ status: 'idle' });
          return;
        }
        setResult({ status: 'loading' });
        let cancelled = false;
        lookupLocation(zip, year).then((r) => {
          if (!cancelled) setResult(r);
        });
        return () => {
          cancelled = true;
        };
      }, [zip, year]);

      return result;
    }
    ```

    export function ZipInput({ value, onChange, error }: ZipInputProps) {
      const handleChange = (e: ChangeEvent<HTMLInputElement>) => {
        // Strip non-digits, cap at 5
        const digits = e.target.value.replace(/\D/g, '').slice(0, 5);
        onChange(digits);
      };
      const errorId = error ? 'zip-error' : undefined;
      return (
        <div className="flex flex-col gap-1.5">
          <Label htmlFor="zip-input">Your ZIP code</Label>
          <Input
            id="zip-input"
            type="text"
            inputMode="numeric"
            pattern="[0-9]{5}"
            placeholder="e.g. 80401"
            value={value}
            onChange={handleChange}
            aria-invalid={error ? 'true' : 'false'}
            aria-describedby={errorId}
            maxLength={5}
          />

          {error && <p id="zip-error" className="text-sm text-red-700">{error}</p>}

    ```typescript
    import { useState, useMemo } from 'react';
    import { useNavigate } from 'react-router';
    import { Button } from '../../ui/Button';
    import { SetupStepLocation } from './SetupStepLocation';
    import { SetupStepPlants } from './SetupStepPlants';
    import { SetupStepReview } from './SetupStepReview';
    import { usePlanStore } from '../../stores/planStore';
    import type { Location } from '../../domain/types';

    type Step = 1 | 2 | 3;

    const STEPS: Array<{ n: Step; label: string; sub: string }> = [
      { n: 1, label: 'Location', sub: 'ZIP and frost dates' },
      { n: 2, label: 'Plants', sub: "Pick what you're growing" },
      { n: 3, label: 'Review', sub: 'Confirm and finish' },
    ];

    export function SetupWizard() {
      const plan = usePlanStore((s) => s.plan);
      const setLocation = usePlanStore((s) => s.setLocation);
      const navigate = useNavigate();

      // Determine initial step ONCE on mount (D-02)
      const initialStep: Step = useMemo(() => (plan === null ? 1 : 2), []);
      // eslint-disable-next-line react-hooks/exhaustive-deps -- intentional: initial step is set-once
      const [step, setStep] = useState<Step>(initialStep);
      const [locationValid, setLocationValid] = useState(false);
      const [pendingLocation, setPendingLocation] = useState<Location | null>(null);

      const handleNext = () => {
        if (step === 1 && pendingLocation) {
          setLocation(pendingLocation);
          setStep(2);
        } else if (step === 2) {
          setStep(3);
        }
      };

      const handleBack = () => {
        if (step > 1) setStep((s) => (s - 1) as Step);
      };

      const handleFinish = () => {
        navigate('/plan');
      };

      const nextDisabled =
        (step === 1 && !locationValid) ||
        (step === 2 && (plan?.plantings.length ?? 0) === 0);

      const showHero = step === 1 && plan === null;

      return (
        <div className="max-w-[var(--spacing-wizard-max-w)] mx-auto px-4 py-12">
          {showHero && (
            <div className="mb-8">
              <h1 className="text-3xl font-semibold text-stone-900">Let&apos;s set up your garden</h1>
              <p className="mt-2 max-w-prose text-base text-stone-600">
                Three quick steps. You can change anything later — nothing&apos;s locked in.
              </p>
            </div>
          )}

          <ol role="list" className="flex items-center justify-between mb-8">
            {STEPS.map(({ n, label, sub }) => {
              const isActive = step === n;
              const isComplete = step > n;
              return (
                <li key={n} aria-current={isActive ? 'step' : undefined} className="flex flex-col items-center">
                  <div
                    className={
                      isActive
                        ? 'h-8 w-8 rounded-full bg-green-700 text-white flex items-center justify-center font-semibold'
                        : isComplete
                        ? 'h-8 w-8 rounded-full bg-green-700 text-white flex items-center justify-center'
                        : 'h-8 w-8 rounded-full border-2 border-stone-200 text-stone-500 flex items-center justify-center font-medium'
                    }
                  >
                    {isComplete ? '✓' : n}
                  </div>
                  <p className={isActive ? 'mt-2 text-sm font-semibold text-stone-900' : 'mt-2 text-sm font-medium text-stone-500'}>{label}</p>
                  <p className="text-xs text-stone-500 hidden md:block">{sub}</p>
                </li>
              );
            })}
          </ol>

          <div className="min-h-[400px]">
            {step === 1 && (
              <SetupStepLocation
                onValidLocation={(loc) => { setPendingLocation(loc); setLocationValid(true); }}
                onLocationInvalid={() => setLocationValid(false)}
              />
            )}
            {step === 2 && <SetupStepPlants />}
            {step === 3 && <SetupStepReview />}
          </div>

          <div className="sticky bottom-0 mt-8 flex items-center justify-between border-t border-stone-200 bg-white py-4">
            {step > 1 ? (
              <Button variant="ghost" onClick={handleBack}>Back</Button>
            ) : (
              <span />
            )}
            {step < 3 ? (
              <Button variant="primary" onClick={handleNext} disabled={nextDisabled}>
                {step === 1 ? 'Next' : 'Next'}
              </Button>
            ) : (
              <Button variant="primary" onClick={handleFinish}>Finish — go to my plan</Button>
            )}
          </div>
        </div>
      );
    }
    ```
