    Step 1 — `src/features/export-reminder/useExportReminder.ts` per RESEARCH §Code Examples:

    ```typescript
    import { useUIStore } from '../../stores/uiStore';
    import { differenceInCalendarDays, parseISO, addDays, format } from 'date-fns';
    import { nowISOString } from '../../domain/dateWrappers';

    const DIRTY_THRESHOLD = 20;
    const AGE_THRESHOLD_DAYS = 14;
    const SNOOZE_LATER_DAYS = 3;
    const SNOOZE_LONG_DAYS = 30;

    export interface ExportReminderState {
      shouldShow: boolean;
      count: number;
      lastExportedAt: string | null;
    }

    export function useShouldShowExportReminder(): ExportReminderState {
      const dirty = useUIStore((s) => s.exportReminder.dirtySinceExport);
      const lastExported = useUIStore((s) => s.exportReminder.lastExportedAt);
      const snoozedUntil = useUIStore((s) => s.exportReminder.snoozedUntil);

      const now = parseISO(nowISOString());

      // Snooze respected first.
      if (snoozedUntil && parseISO(snoozedUntil) > now) {
        return { shouldShow: false, count: dirty, lastExportedAt: lastExported };
      }

      // No edits = no nagging (D-12 explicit).
      if (dirty <= 0) return { shouldShow: false, count: dirty, lastExportedAt: lastExported };

      // Trigger A: dirty count threshold.
      if (dirty >= DIRTY_THRESHOLD) {
        return { shouldShow: true, count: dirty, lastExportedAt: lastExported };
      }

      // Trigger B: age + dirty>0.
      if (lastExported) {
        const daysSince = differenceInCalendarDays(now, parseISO(lastExported));
        if (daysSince >= AGE_THRESHOLD_DAYS) {
          return { shouldShow: true, count: dirty, lastExportedAt: lastExported };
        }
      }

      return { shouldShow: false, count: dirty, lastExportedAt: lastExported };
    }

    /** Snooze action helpers — used by banner buttons. */
    export function useExportReminder() {
      const setSnoozedUntil = useUIStore((s) => s.setSnoozedUntil);
      const state = useShouldShowExportReminder();
      const snoozeNDays = (n: number) => {
        setSnoozedUntil(addDays(parseISO(nowISOString()), n).toISOString());
      };
      return {
        ...state,
        snooze3Days: () => snoozeNDays(SNOOZE_LATER_DAYS),
        snooze30Days: () => snoozeNDays(SNOOZE_LONG_DAYS),
        formatLastExportedShort: () => state.lastExportedAt
          ? format(parseISO(state.lastExportedAt), 'MMM d')
          : 'you started',
      };
    }
    ```

    NOTE on raw `new Date()`: ESLint allowlist for `no-restricted-syntax`. The helpers above use `parseISO` from date-fns + `addDays(...).toISOString()` — no raw `new Date(string)`. The `nowISOString()` from dateWrappers is the only authority for "now". `addDays(parseISO(...), n).toISOString()` does NOT trigger the lint rule (the rule targets `new Date(arg)` form).

## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| user export action → uiStore reset | exportPlan side-effect resets dirtySinceExport + records lastExportedAt; snooze writes snoozedUntil ISO |
| planStore mutation → uiStore.incrementDirty | one-way dependency (planStore → uiStore); circular risk if uiStore ever imports planStore (Plan 04 confirmed it does NOT) |
| date-fns parseISO → trigger evaluation | malformed ISO in localStorage envelope could throw |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-04-05-01 | Tampering | Malicious user edits localStorage to set snoozedUntil far in future | accept | Single-user local app; user has full control of their own state. Worst case: user disables their own reminder. |
| T-04-05-02 | Repudiation | Counter accidentally double-counts because zundo replay invokes setter | mitigate | zundo replays state directly (per Phase 3 D-15 / RESEARCH §Architecture); does NOT re-invoke setters. Test 11 in Task 1 explicitly verifies undo/redo do not increment counter. |
| T-04-05-03 | Information Disclosure | dirtySinceExport reveals user activity rate to anyone with localStorage access | accept | Same threat profile as existing planStore data; localStorage isolated per origin. |
| T-04-05-04 | Denial of Service | Malformed snoozedUntil ISO causes parseISO throw on render | mitigate | parseISO returns Invalid Date for malformed input; comparison `> now` returns false (Invalid Date comparison is false in JS). Effectively treats malformed snoozedUntil as expired snooze — banner shows. Acceptable degradation. Could harden with try/catch in v2. |
| T-04-05-05 | Elevation of Privilege | exportPlan failure resets counter spuriously | mitigate | exportPlan side-effects ONLY on the success branch (return ok=true). Test in Task 1 covers the failure path. |
</threat_model>
