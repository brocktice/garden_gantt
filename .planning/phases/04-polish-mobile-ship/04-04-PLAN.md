    Step 2 — `src/features/onboarding/useCoachMarks.ts`:
    ```typescript
    import { useState, useMemo } from 'react';
    import { useUIStore } from '../../stores/uiStore';
    import { usePlanStore } from '../../stores/planStore';
    import { MARKS, type CoachMark } from './coachMarks.types';

    export interface UseCoachMarksResult {
      active: boolean;
      currentMark: CoachMark | null;
      currentIndex: number;
      totalCount: number;
      isLast: boolean;
      dismiss: () => void;
      advance: () => void;
    }

    export function useCoachMarks(currentRoute: string): UseCoachMarksResult {
      const dismissed = useUIStore((s) => s.onboarding.coachMarksDismissed);
      const setDismissed = useUIStore((s) => s.setCoachMarksDismissed);
      const plantingCount = usePlanStore((s) => s.plan?.plantings.length ?? 0);
      const [currentIndex, setCurrentIndex] = useState(0);

      const onPlanRoute = currentRoute === '/plan' || currentRoute.startsWith('/plan?');
      const visibleMarks = useMemo(
        () => MARKS.filter((m) => !m.requiresPlantings || plantingCount > 0),
        [plantingCount],
      );

      const inactive = dismissed || !onPlanRoute || visibleMarks.length === 0;
      const safeIndex = Math.min(currentIndex, visibleMarks.length - 1);
      const currentMark = inactive ? null : visibleMarks[safeIndex] ?? null;

      const dismiss = () => setDismissed(true);
      const advance = () => {
        if (safeIndex >= visibleMarks.length - 1) {
          dismiss(); // [Got it] on last mark
        } else {
          setCurrentIndex(safeIndex + 1);
        }
      };

      return {
        active: !inactive,
        currentMark,
        currentIndex: safeIndex,
        totalCount: visibleMarks.length,
        isLast: safeIndex === visibleMarks.length - 1,
        dismiss,
        advance,
      };
    }
    ```

    Step 1 — `src/features/onboarding/CoachMarks.tsx` per PATTERNS analog (ConstraintTooltip.tsx) + UI-SPEC §Coach-mark visual style:

    ```tsx
    import { useEffect, useLayoutEffect, useState } from 'react';
    import { createPortal } from 'react-dom';
    import { useLocation } from 'react-router';
    import { Button } from '../../ui/Button';
    import { useCoachMarks } from './useCoachMarks';

    const CALLOUT_W = 280; // matches --spacing-coach-mark-callout
    const CALLOUT_GAP = 12; // px below target

    export function CoachMarks() {
      const location = useLocation();
      const { active, currentMark, currentIndex, totalCount, isLast, dismiss, advance } = useCoachMarks(location.pathname);
      const [anchorPos, setAnchorPos] = useState<{ left: number; top: number } | null>(null);

      // Position callout via target's getBoundingClientRect
      useLayoutEffect(() => {
        if (!active || !currentMark) {
          setAnchorPos(null);
          return;
        }
        const compute = () => {
          const el = document.querySelector(`[data-coach-target="${currentMark.id}"]`);
          if (!el) { setAnchorPos(null); return; }
          const r = el.getBoundingClientRect();
          // Prefer below-target; fall back above if no room
          const belowTop = r.bottom + CALLOUT_GAP;
          setAnchorPos({
            left: Math.max(8, r.left + r.width / 2 - CALLOUT_W / 2),
            top: belowTop,
          });
        };
        queueMicrotask(compute);
        window.addEventListener('resize', compute);
        window.addEventListener('scroll', compute, true);
        return () => {
          window.removeEventListener('resize', compute);
          window.removeEventListener('scroll', compute, true);
        };
      }, [active, currentMark]);

      // Esc to dismiss, Enter to advance (per UI-SPEC §Onboarding coach marks)
      useEffect(() => {
        if (!active) return;
        const onKey = (e: KeyboardEvent) => {
          // Don't intercept when typing in form fields
          const t = e.target;
          if (t instanceof HTMLElement && t.matches('input, textarea, [contenteditable]')) return;
          if (e.key === 'Escape') { e.preventDefault(); dismiss(); }
          else if (e.key === 'Enter') { e.preventDefault(); advance(); }
        };
        document.addEventListener('keydown', onKey);
        return () => document.removeEventListener('keydown', onKey);
      }, [active, dismiss, advance]);

      if (!active || !currentMark || !anchorPos) return null;

      return createPortal(
        <>
          {/* Backdrop */}
          <div
            className="fixed inset-0 z-40 bg-stone-900/40"
            onClick={dismiss} // click outside dismisses (UX nicety)
          />
          {/* Callout */}
          <div
            role="dialog"
            aria-modal="true"
            aria-labelledby="coach-mark-heading"
            className="fixed z-50 bg-white rounded-md border border-stone-200 shadow-lg p-4"
            style={{
              left: anchorPos.left,
              top: anchorPos.top,
              maxWidth: 'var(--spacing-coach-mark-callout)',
            }}
          >
            <div className="flex items-start justify-between gap-2 mb-2">
              <h2 id="coach-mark-heading" className="text-sm font-medium text-stone-900">{currentMark.headingLabel}</h2>
              <span className="text-sm font-medium text-stone-600">{currentIndex + 1} of {totalCount}</span>
            </div>
            <p className="text-base text-stone-700 mb-4">{currentMark.body}</p>
            <div className="flex justify-end gap-2">
              <Button variant="ghost" onClick={dismiss}>Skip tour</Button>
              <Button variant="primary" onClick={advance}>{isLast ? 'Got it' : 'Next →'}</Button>
            </div>
          </div>
        </>,
        document.body,
      );
    }
    ```
