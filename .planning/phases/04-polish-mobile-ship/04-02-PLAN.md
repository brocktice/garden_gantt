    /**
     * Convert YYYY-MM-DD (e.g. from <input type=date>) to UTC-noon ISO string.
     * Pinned to noon UTC to dodge DST/timezone shifts (SCH-03 invariant).
     */
    export function ymdToISONoon(ymd: string): string {
      // Inline string concat — no `new Date()` call, preserves UTC-noon discipline.

    }

    /**
     * Inverse — extract YYYY-MM-DD from a UTC-noon ISO string for <input type=date> defaultValue.

    export function isoNoonToYMD(iso: string): string {

    }
